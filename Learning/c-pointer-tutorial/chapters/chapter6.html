Pointers and Arrays

陣列與指標的等價關係、記憶體配置、指標運算與存取方式

1 陣列記憶體配置基礎

陣列是在記憶體中連續配置的一系列相同型別元素。每個 int 佔 4 bytes，所以 int A[5] 總共佔用 20 bytes。

Index A[0] A[1] A[2] A[3] A[4]
Value 2 4 6 8 2
Address 200 204 208 212 216 
// array_memory_demo.c
#include <stdio.h>

int main(void) {
    // int → 4 bytes
    // int A[5] → 5 × 4 bytes = 20 bytes
    int A[5] = {2, 4, 6, 8, 2};
    
    printf("A 代表陣列的基底位址 (base address)\n");
    printf("A     = %p\n", (void*)A);
    printf("&A[0] = %p\n", (void*)&A[0]);
    printf("&A[1] = %p\n", (void*)&A[1]);
    printf("&A[2] = %p\n", (void*)&A[2]);
    printf("&A[3] = %p\n", (void*)&A[3]);
    printf("&A[4] = %p\n", (void*)&A[4]);
    
    printf("\n每個元素相差 %zu bytes\n", sizeof(int));
    
    return 0;
}

2 陣列名稱與指標的關係

陣列名稱 A 實際上代表指向陣列第一個元素的指標，也就是 &A[0]。我們可以透過指標來存取陣列元素。 

// array_pointer_demo.c
#include <stdio.h>

int main(void) {
    int A[5] = {2, 4, 6, 8, 2};
    int *P;
    
    // P 指向陣列 A 的第一個元素
    P = A;  // 等價於 P = &A[0]
    
    printf("A     = %p\n", (void*)A);
    printf("P     = %p\n", (void*)P);
    printf("&A[0] = %p\n", (void*)&A[0]);
    
    printf("\n印出 A (印出位址，例如 200):\n");
    printf("printf(\"%%d\\n\", A);     // 印出 %p\n", (void*)A);
    
    printf("\n印出 *A (A[0] 的值，例如 2):\n");
    printf("printf(\"%%d\\n\", *A);    // 印出 %d\n", *A);
    
    printf("\n印出 A+1 (下一個位址，例如 204):\n");
    printf("printf(\"%%d\\n\", A+1);   // 印出 %p\n", (void*)(A+1));
    
    printf("\n印出 *(A+1) (A[1] 的值，例如 4):\n");
    printf("printf(\"%%d\\n\", *(A+1)); // 印出 %d\n", *(A+1));
    
    return 0;
}
📌 重要概念

A 和 &A[0] 是等價的，都代表陣列第一個元素的位址。

3 元素存取方式

陣列元素有兩種等價的存取方式：透過陣列索引或透過指標運算。

🔍 存取方式對照表
  • 位址 (Address) → &A[i] 或 (A + i)
  • 值 (Value) → A[i] 或 *(A + i)
// access_methods_demo.c
#include <stdio.h>

int main(void) {
    int A[5] = {2, 4, 6, 8, 2};
    
    printf("陣列元素存取的兩種等價方式：\n\n");
    
    // 存取第 i 個元素 (i=1)
    int i = 1;
    
    printf("Element at index %d:\n", i);
    printf("Address: &A[%d] = %p  or  (A + %d) = %p\n", 
           i, (void*)&A[i], i, (void*)(A + i));
    printf("Value:   A[%d]  = %d   or  *(A + %d) = %d\n\n", 
           i, A[i], i, *(A + i));
    
    // 用指標方式印出所有元素
    printf("用指標方式印出所有元素：\n");
    for (int j = 0; j < 5; j++) {
        printf("A[%d] = %d, 位址 = %p\n", 
               j, *(A + j), (void*)(A + j));
    }
    
    return 0;
}

4 指標運算詳解

當我們對指標進行加法運算時，實際增加的位址是該型別的大小。例如 A + 1 實際上是 A + 1 × sizeof(int)。 

// pointer_arithmetic_demo.c
#include <stdio.h>

int main(void) {
    int A[5] = {2, 4, 6, 8, 2};
    
    printf("指標運算示範：\n");
    printf("sizeof(int) = %zu bytes\n\n", sizeof(int));
    
    // 印出每個位址與值
    for (int i = 0; i < 5; i++) {
        printf("A + %d = %p, *(A + %d) = %d\n", 
               i, (void*)(A + i), i, *(A + i));
    }
    
    printf("\n位址差值計算：\n");
    printf("(A + 1) - A = %ld (相差 %zu bytes)\n", 
           (A + 1) - A, (long)((A + 1) - A) * sizeof(int));
    printf("(A + 2) - A = %ld (相差 %zu bytes)\n", 
           (A + 2) - A, (long)((A + 2) - A) * sizeof(int));
    
    return 0;
}

5 透過指標修改陣列元素

我們可以透過指標運算來修改陣列的元素，這與直接使用陣列索引是等價的。 

// modify_array_demo.c
#include <stdio.h>

void print_array(int arr[], int size) {
    printf("陣列內容: [");
    for (int i = 0; i < size; i++) {
        printf("%d", arr[i]);
        if (i < size - 1) printf(", ");
    }
    printf("]\n");
}

int main(void) {
    int A[5] = {2, 4, 6, 8, 2};
    
    printf("原始陣列：\n");
    print_array(A, 5);
    
    // 透過指標運算修改元素
    printf("\n透過指標運算修改元素：\n");
    *(A + 0) = 10;   // 等價於 A[0] = 10
    *(A + 2) = 30;   // 等價於 A[2] = 30
    *(A + 4) = 50;   // 等價於 A[4] = 50
    
    print_array(A, 5);
    
    // 用指標變數修改
    printf("\n用指標變數修改：\n");
    int *P = A;
    *P = 100;        // 修改 A[0]
    *(P + 1) = 200; // 修改 A[1]
    
    print_array(A, 5);
    
    return 0;
}

6 陣列與指標的重要區別

雖然陣列名稱可以當作指標使用，但它們有一些重要的差異：

  • 陣列名稱：是常數指標，不能被重新賦值
  • 指標變數：可以指向不同的位址
  • sizeof 運算：對陣列名稱會回傳整個陣列的大小
// array_vs_pointer_demo.c
#include <stdio.h>

int main(void) {
    int A[5] = {2, 4, 6, 8, 2};
    int B[3] = {1, 3, 5};
    int *P;
    
    printf("陣列與指標的區別：\n\n");
    
    // sizeof 運算的差異
    printf("sizeof(A) = %zu bytes (整個陣列)\n", sizeof(A));
    printf("sizeof(P) = %zu bytes (指標本身)\n\n", sizeof(P));
    
    // 指標可以重新賦值
    P = A;
    printf("P = A; P 現在指向陣列 A\n");
    printf("*P = %d (A[0])\n\n", *P);
    
    P = B;
    printf("P = B; P 現在指向陣列 B\n");
    printf("*P = %d (B[0])\n\n", *P);
    
    // 陣列名稱不能重新賦值 (會編譯錯誤)
    // A = B;  // 錯誤！陣列名稱是常數
    
    printf("注意：陣列名稱 A 不能被重新賦值\n");
    printf("A 永遠指向 &A[0] = %p\n", (void*)A);
    
    return 0;
}

✅ 本章重點總結

🎯 核心概念

陣列與指標密不可分，理解它們的等價關係是掌握 C 語言記憶體操作的關鍵。

  • 陣列記憶體配置：連續配置，每個元素相差一個型別大小的位址。
  • 陣列名稱等價：A == &A[0]，都代表陣列第一個元素的位址。
  • 存取方式等價：A[i] == *(A + i)，&A[i] == (A + i)。
  • 指標運算：A + i 實際增加 i × sizeof(type) 個位元組。
  • 陣列 vs 指標：陣列名稱是常數指標，不能重新賦值；指標變數可以改變指向。
